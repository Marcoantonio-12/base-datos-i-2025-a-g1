CREATE TABLE PERSONA (
    id INT PRIMARY KEY AUTO_INCREMENT,
    nombre VARCHAR(100) NOT NULL,
    apellido VARCHAR(100) NOT NULL,
    fecha_nacimiento DATE NOT NULL,
    correo VARCHAR(100) UNIQUE NOT NULL,
    direccion VARCHAR(255),
    telefono VARCHAR(20)
);

CREATE TABLE CLIENTE (
    id INT PRIMARY KEY AUTO_INCREMENT,
    codigo VARCHAR(50) UNIQUE NOT NULL,
    tipo_vinculacion VARCHAR(50) NOT NULL,
    persona_id INT NOT NULL,
    FOREIGN KEY (persona_id) REFERENCES PERSONA(id) ON DELETE CASCADE
);

CREATE TABLE EMPLEADO (
    id INT PRIMARY KEY AUTO_INCREMENT,
    codigo VARCHAR(50) UNIQUE NOT NULL,
    tipo_vinculacion VARCHAR(50) NOT NULL,
    salario DOUBLE NOT NULL,
    tipo_contrato VARCHAR(50) NOT NULL,
    persona_id INT NOT NULL,
    FOREIGN KEY (persona_id) REFERENCES PERSONA(id) ON DELETE CASCADE
);

CREATE TABLE CATEGORIA (
    id INT PRIMARY KEY AUTO_INCREMENT,
    nombre VARCHAR(100) NOT NULL,
    descripcion TEXT
);

CREATE TABLE METODO_PAGO (
    id INT PRIMARY KEY AUTO_INCREMENT,
    nombre VARCHAR(100) NOT NULL,
    descripcion TEXT
);

CREATE TABLE PRODUCTO (
    id INT PRIMARY KEY AUTO_INCREMENT,
    codigo VARCHAR(50) UNIQUE NOT NULL,
    descripcion TEXT,
    categoria_id INT NOT NULL,
    FOREIGN KEY (categoria_id) REFERENCES CATEGORIA(id) ON DELETE SET NULL
);

CREATE TABLE INVENTARIO (
    id INT PRIMARY KEY AUTO_INCREMENT,
    nombre VARCHAR(100) NOT NULL,
    fecha DATE NOT NULL,
    producto_id INT NOT NULL,
    stock INT NOT NULL DEFAULT 0,
    fecha_lote DATE,
    fecha_vencimiento DATE,
    FOREIGN KEY (producto_id) REFERENCES PRODUCTO(id) ON DELETE CASCADE
);

CREATE TABLE FACTURA (
    id INT PRIMARY KEY AUTO_INCREMENT,
    fecha DATE NOT NULL,
    valor_neto DOUBLE NOT NULL,
    valor_descuento DOUBLE DEFAULT 0,
    valor_incremento DOUBLE DEFAULT 0,
    cliente_id INT NOT NULL,
    medio_pago_id INT NOT NULL,
    FOREIGN KEY (cliente_id) REFERENCES CLIENTE(id) ON DELETE CASCADE,
    FOREIGN KEY (medio_pago_id) REFERENCES METODO_PAGO(id) ON DELETE SET NULL
);

CREATE TABLE DETALLE_FACTURA (
    id INT PRIMARY KEY AUTO_INCREMENT,
    cantidad INT NOT NULL,
    porcentaje_descuento DOUBLE DEFAULT 0,
    porcentaje_incremento DOUBLE DEFAULT 0,
    subtotal DOUBLE NOT NULL,
    producto_id INT NOT NULL,
    factura_id INT NOT NULL,
    FOREIGN KEY (producto_id) REFERENCES PRODUCTO(id) ON DELETE CASCADE,
    FOREIGN KEY (factura_id) REFERENCES FACTURA(id) ON DELETE CASCADE
);
